017public class Location {
018        
019    // The rough distance that this earthquake occurred away from the reporting station, measured in degrees between. 1 degree is approximately 111.2 kilometers. In general, the smaller this number, the more reliable is the calculated depth of the earthquake. In general, this number is between 0.4-7.1.
020    private Double distance;
021    private String full;
022    // A best guess for the name of the state (or country, in some cases) that this earthquake was reported in.
023    private String name;
024    // Decimal degrees longitude (east and west on the globe). Negative values for western latitudes. Ranges from -180 to 180.
025    private Double longitude;
026    // Depth of the event in kilometers.
027    private Double depth;
028    // Decimal degrees latitude (up and down on the globe). Negative values for southern latitudes. Ranges from -90 to 90.
029    private Double latitude;
030    
031    
032    /*
033     * @return 
034     */
035    public Double getDistance() {
036        return this.distance;
037    }
038    
039    
040    
041    /*
042     * @return 
043     */
044    public String getFull() {
045        return this.full;
046    }
047    
048    
049    
050    /*
051     * @return 
052     */
053    public String getName() {
054        return this.name;
055    }
056    
057    
058    
059    /*
060     * @return 
061     */
062    public Double getLongitude() {
063        return this.longitude;
064    }
065    
066    
067    
068    /*
069     * @return 
070     */
071    public Double getDepth() {
072        return this.depth;
073    }
074    
075    
076    
077    /*
078     * @return 
079     */
080    public Double getLatitude() {
081        return this.latitude;
082    }
083    
084    
085    
086        
087        /**
088         * Creates a string based representation of this Location.
089        
090         * @return String
091         */
092        public String toString() {
093                return "Location[" +distance+", "+full+", "+name+", "+longitude+", "+depth+", "+latitude+"]";
094        }
095        
096        /**
097         * Internal constructor to create a Location from a  representation.
098         * @param json_data The raw json data that will be parsed.
099         */
100    public Location(JSONObject json_data) {
101        try {// distance
102            this.distance = (Double)json_data.get("distance");// full
103            this.full = (String)json_data.get("full");// name
104            this.name = (String)json_data.get("name");// longitude
105            this.longitude = (Double)json_data.get("longitude");// depth
106            this.depth = (Double)json_data.get("depth");// latitude
107            this.latitude = (Double)json_data.get("latitude");
108        } catch (NullPointerException e) {
109                System.err.println("Could not convert the response to a Location; a field was missing.");
110                e.printStackTrace();
111        } catch (ClassCastException e) {
112                System.err.println("Could not convert the response to a Location; a field had the wrong structure.");
113                e.printStackTrace();
114        }
115        }       
116}
