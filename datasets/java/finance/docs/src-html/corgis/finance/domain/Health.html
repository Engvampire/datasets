070            this.healthTotalExpenditure = ((Number)json_data.get("Health Total Expenditure")).intValue();// Hospitals
071            this.hospitals = new Hospitals((JSONObject)json_data.get("Hospitals"));// Health Intergovernmental
072            this.healthIntergovernmental = ((Number)json_data.get("Health Intergovernmental")).intValue();

076        } catch (ClassCastException e) {
