149            this.naturalResourcesEquipmentAndLand = new Integer(((Long)json_data.get("Natural Resources Equipment and Land")).intValue());// Agriculture
150            this.agriculture = new Agriculture((JSONObject)json_data.get("Agriculture"));
151        } catch (NullPointerException e) {
152                System.err.println("Could not convert the response to a NaturalResources; a field was missing.");
153                e.printStackTrace();
154        } catch (ClassCastException e) {
