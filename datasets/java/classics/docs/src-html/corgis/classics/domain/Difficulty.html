019    // The 'Flesch Reading Ease' uses the sentence length (number of words per sentence) and the number of syllables per word in an equation to calculate the reading ease. Texts with a very high Flesch reading Ease score (about 100) are very easy to read, have short sentences and no words of more than two syllables.
020    private Double fleschReadingEase;
021    // The Automated Readability Index is a number indicating the understandability of the text. This number is an approximate US Grade Level needed to comprehend the text, calculated using the characters per word and words per sentences.
022    private Double automatedReadabilityIndex;
023    // The Coleman Liau Index is a number indicating the understandability of the text. This number is an approximate US Grade Level needed to comprehend the text, calculated using characters instead of syllables, similar to the Automated Readability Index.
024    private Double colemanLiauIndex;
025    // The Gunning Fog Index measures the readability of English writing. The index estimates the years of formal education needed to understand the text on a first reading. The formula is calculated using the ratio of words to sentences and the percentage of words that are complex (i.e. have three or more syllables).
026    private Double gunningFog;
027    // Linsear Write is a readability metric for English text, purportedly developed for the United States Air Force to help them calculate the readability of their technical manuals. It was designed to calculate the United States grade level of a text sample based on sentence length and the number words used that have three or more syllables.
028    private Double linsearWriteFormula;
029    // The Dale Chall Readability Score provides a numeric gauge of the comprehension difficulty that readers come upon when reading a text. It uses a list of 3000 words that groups of fourth-grade American students could reliably understand, considering any word not on that list to be difficult. This number is an approximate US Grade Level needed to comprehend the text.
030    private Double daleChallReadabilityScore;
031    // The "Flesch-Kincaid Grade Level Formula" presents a score as a U.S. grade level, making it easier to understand. It uses a similar formula to the Flesch Reading Ease measure.
032    private Double fleschKincaidGrade;
033    // The SMOG grade is a measure of readability that estimates the years of education needed to understand a piece of writing. SMOG is the acronym derived from "Simple Measure of Gobbledygook". Its formula is based on the number of polysyllables (words with three or more syllables) and the number of sentences.
034    private Double smogIndex;
035    // The number of words in the text that are considered "difficult"; that is, they are not on a list of 3000 words that are considered understandable by fourth-grade American students.
036    private Integer difficultWords;

130        /**
131         * Internal constructor to create a Difficulty from a  representation.
132         * @param json_data The raw json data that will be parsed.
133         */
134    public Difficulty(JSONObject json_data) {
135        try {// flesch reading ease
136            this.fleschReadingEase = (Double)json_data.get("flesch reading ease");// automated readability index
137            this.automatedReadabilityIndex = (Double)json_data.get("automated readability index");// coleman liau index
138            this.colemanLiauIndex = (Double)json_data.get("coleman liau index");// gunning fog
139            this.gunningFog = (Double)json_data.get("gunning fog");// linsear write formula
140            this.linsearWriteFormula = (Double)json_data.get("linsear write formula");// dale chall readability score
141            this.daleChallReadabilityScore = (Double)json_data.get("dale chall readability score");// flesch kincaid grade
142            this.fleschKincaidGrade = (Double)json_data.get("flesch kincaid grade");// smog index
143            this.smogIndex = (Double)json_data.get("smog index");// difficult words
144            this.difficultWords = new Integer(((Long)json_data.get("difficult words")).intValue());
145        } catch (NullPointerException e) {
146                System.err.println("Could not convert the response to a Difficulty; a field was missing.");
147                e.printStackTrace();
148        } catch (ClassCastException e) {
149                System.err.println("Could not convert the response to a Difficulty; a field had the wrong structure.");
150                e.printStackTrace();
151        }
