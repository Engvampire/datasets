069        this.connection = null;
070        try {
071            this.connection = DriverManager.getConnection("jdbc:sqlite:"+databasePath);
072        } catch ( Exception e ) {
073            System.err.println( e.getClass().getName() + ": " + e.getMessage() );
074            System.exit(0);
075        }
076        this.parser =  new JSONParser();
077    }

109    /**
110     * Returns information about the status of disease for the given year, from 1928 to 2011.
111    
112     * @param year A year between 1928 and 2011.
113     * @return a list[report]
114     */
115        public ArrayList<Report> getReportsByYear(Integer year, boolean test) {
116        String query;
117        if (test) {
118            query = String.format("SELECT data FROM health WHERE year=? ORDER BY year, location, disease ASC LIMIT %d", this.HARDWARE);
119        } else {
120            query = "SELECT data FROM health WHERE year=? ORDER BY year, location, disease ASC";
121        }
122        PreparedStatement preparedQuery = null;
123        ResultSet rs = null;
124        try {
125            preparedQuery = this.connection.prepareStatement(query);
126        } catch (SQLException e) {
127            System.err.println("Could not build SQL query for local database.");
128                e.printStackTrace();
129        }
130        try {
131            preparedQuery.setInt(1, year);
132        } catch (SQLException e) {
133            System.err.println("Could not build prepare argument: year");
134                e.printStackTrace();
135        }
136        try {
137            rs = preparedQuery.executeQuery();
138        } catch (SQLException e) {
139            System.err.println("Could not execute query.");
140                e.printStackTrace();
141        }
142        
143        ArrayList<Report> result = new ArrayList<Report>();
144        try {
145            while (rs.next()) {
146                String raw_result = rs.getString(1);
147                Report parsed = null;
148                if (test) {
149                    parsed = new Report(((JSONObject)this.parser.parse(raw_result)));
150                    
151                } else {
152                    parsed = new Report(((JSONObject)this.parser.parse(raw_result)));
153                    
154                }
155                
156                result.add(parsed);
157                
158            }
159        } catch (SQLException e) {
160            System.err.println("Could not iterate through query.");
161                e.printStackTrace();
162        } catch (ParseException e) {
163            System.err.println("Could not convert the response from getReportsByYear; a parser error occurred.");
164                e.printStackTrace();
165        }
166        return result;
167        }
