069    private void connectToDatabase(String databasePath) {
070        this.connection = null;
071        try {
072            this.connection = DriverManager.getConnection("jdbc:sqlite:"+databasePath);
073        } catch ( Exception e ) {
074            System.err.println( e.getClass().getName() + ": " + e.getMessage() );
075            System.exit(0);
076        }
077        this.parser =  new JSONParser();
078    }

110    /**
111     * Returns information about the status of disease for the given year, from 1928 to 2011.
112    
113     * @param year A year between 1928 and 2011.
114     * @return a list[report]
115     */
116        public ArrayList<Report> getReportsByYear(Integer year, boolean test) {
117        String query;
118        if (test) {
119            query = String.format("SELECT data FROM health WHERE year=? ORDER BY year, location, disease ASC LIMIT %d", this.HARDWARE);
120        } else {
121            query = "SELECT data FROM health WHERE year=? ORDER BY year, location, disease ASC";
122        }
123        PreparedStatement preparedQuery = null;
124        ResultSet rs = null;
125        try {
126            preparedQuery = this.connection.prepareStatement(query);
127        } catch (SQLException e) {
128            System.err.println("Could not build SQL query for local database.");
129                e.printStackTrace();
130        }
131        try {
132            preparedQuery.setInt(1, year);
133        } catch (SQLException e) {
134            System.err.println("Could not build prepare argument: year");
135                e.printStackTrace();
136        }
137        try {
138            rs = preparedQuery.executeQuery();
139        } catch (SQLException e) {
140            System.err.println("Could not execute query.");
141                e.printStackTrace();
142        }
143        
144        ArrayList<Report> result = new ArrayList<Report>();
145        try {
146            while (rs.next()) {
147                String raw_result = rs.getString(1);
148                Report parsed = null;
149                if (test) {
150                    parsed = new Report(((JSONObject)this.parser.parse(raw_result)));
151                    
152                } else {
153                    parsed = new Report(((JSONObject)this.parser.parse(raw_result)));
154                    
155                }
156                
157                result.add(parsed);
158                
159            }
160        } catch (SQLException e) {
161            System.err.println("Could not iterate through query.");
162                e.printStackTrace();
163        } catch (ParseException e) {
164            System.err.println("Could not convert the response from getReportsByYear; a parser error occurred.");
165                e.printStackTrace();
166        }
167        return result;
168        }
