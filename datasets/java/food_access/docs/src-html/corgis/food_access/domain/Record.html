020 */
021public class Record {
022        
023    // Numbers are summation of the Census Tracts for each county, distances refer to distance that an individual has to travel to access supermarkets, supercenters, grocery stores, or other sources of healthy and affordable food.
024    private LowAccessNumbers lowAccessNumbers;
025    // Housing units without vehicle access and low access as specified distance
026    private VehicleAccess vehicleAccess;
027    private HousingData housingData;
028    private String county;
029    // Percents are number of Census Tracts with listed classification divided by total tracts for the county
030    private LowAccessPercents lowAccessPercents;
031    private String state;
032    private Integer population;
033    
034    
035    /*
036     * @return

038    public LowAccessNumbers getLowAccessNumbers() {

040    }

043    
044    /*

047    public VehicleAccess getVehicleAccess() {
048        return this.vehicleAccess;
049    }

051    
052    
053    /*
054     * @return 
055     */
056    public HousingData getHousingData() {
057        return this.housingData;
058    }
059    
060    
061    
062    /*
063     * @return 
064     */
065    public String getCounty() {
066        return this.county;
067    }

094    }
095    
096    
097    
098        
099        /**
100         * Creates a string based representation of this Record.
101        
102         * @return String
103         */
104        public String toString() {
105                return "Record[" +lowAccessNumbers+", "+vehicleAccess+", "+housingData+", "+county+", "+lowAccessPercents+", "+state+", "+population+"]";
106        }
107        
108        /**
109         * Internal constructor to create a Record from a  representation.
110         * @param json_data The raw json data that will be parsed.
111         */
112    public Record(JSONObject json_data) {
113        try {// Low Access Numbers
114            this.lowAccessNumbers = new LowAccessNumbers((JSONObject)json_data.get("Low Access Numbers"));// Vehicle Access
115            this.vehicleAccess = new VehicleAccess((JSONObject)json_data.get("Vehicle Access"));// Housing Data
116            this.housingData = new HousingData((JSONObject)json_data.get("Housing Data"));// County
117            this.county = (String)json_data.get("County");// Low Access Percents
118            this.lowAccessPercents = new LowAccessPercents((JSONObject)json_data.get("Low Access Percents"));// State
119            this.state = (String)json_data.get("State");// Population
120            this.population = new Integer(((Long)json_data.get("Population")).intValue());
121        } catch (NullPointerException e) {
122                System.err.println("Could not convert the response to a Record; a field was missing.");
123                e.printStackTrace();
124        } catch (ClassCastException e) {
125                System.err.println("Could not convert the response to a Record; a field had the wrong structure.");
126                e.printStackTrace();
127        }
128        }       
129}
